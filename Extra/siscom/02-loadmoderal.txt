function setViews($topicoId){
	$topicoId = mysql_real_escape_string($topicoId);
	$readArtigo = read('posts',"WHERE id = '$topicoId'");
	
	foreach($readArtigo as $artigo);
		$views = $artigo['views'];
		$views = $views +1;
		$dataViews = array(
			'views' => $views
		);
		update('posts',$dataViews,"id = '$topicoId'");
}
